// Kth Smallest Element in a BST

// Approach 1
class Solution {
    private void small(TreeNode root, List<Integer> list){
        if(root == null) return;
        small(root.left, list);
        list.add(root.val);
        small(root.right, list);
    }
    public int kthSmallest(TreeNode root, int k) {
        List<Integer> list = new ArrayList<>();
        small(root, list);
        if(k <= 0 || k > list.size()) return -1;
        return list.get(k - 1);
    }
}

// Approach 2
class Solution {
    private int count = 0;
    private int res = -1;
    private void small(TreeNode root, int k){
        if(root == null) return;
        small(root.left, k);
        count++;
        if(count == k){
            res = root.val;
            return;
        }
        small(root.right, k);
    }
    public int kthSmallest(TreeNode root, int k) {
        count = 0;
        res = -1;
        small(root, k);
        return res;
    }
}
