// Removing Stars from a String

//Approach 1
class Solution {
    public String removeStars(String s) {
        Stack<Character> stack = new Stack<>();
        for(Character ch : s.toCharArray()) {
            if(ch == '*') stack.pop();
            else stack.push(ch);
        }

        StringBuilder sb = new StringBuilder();
        for(Character ch : stack) sb.append(ch);

        return sb.toString();
    }
}

//Approach 2
class Solution {
    public String removeStars(String s) {
        StringBuilder sb = new StringBuilder();
        for(Character ch : s.toCharArray()) {
            if(ch == '*') sb.deleteCharAt(sb.length() - 1);
            else sb.append(ch);
        }

        return sb.toString();
    }
}

//Approach 3
class Solution {
    public String removeStars(String s) {
        char[] ch = new char[s.length()];
        int j = 0;
        for(int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            if(c == '*'){
                j--;
                continue;
            }
            ch[j++] = c;
        }

        return new String(ch, 0, j);
    }
}
