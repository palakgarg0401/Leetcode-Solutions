// Contiguous Array

class Solution {
    public int findMaxLength(int[] arr) {
        Map<Integer, Integer> map = new HashMap<>();
        map.put(0, -1);
        int sum = 0, max = 0;
        for(int i = 0; i < arr.length; i++){
            sum += (arr[i] == 0) ? -1 : 1;

            if(map.containsKey(sum)){
                int len = i - map.get(sum);
                max = Math.max(max, len);
            }
            else map.put(sum, i);
        }
        return max;
    }
}
