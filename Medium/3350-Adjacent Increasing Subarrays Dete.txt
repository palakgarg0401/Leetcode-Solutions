// Adjacent Increasing Subarrays Detection II

class Solution {
    public int maxIncreasingSubarrays(List<Integer> nums) {
        int max = 0, prev = 0, curr = 0;
        int n = nums.size();

        for(int i = 0; i < n; i++){
            curr++;
            if(i == n - 1 || nums.get(i) >= nums.get(i+1)){
                max = Math.max(max, Math.max(curr/2, Math.min(prev, curr)));

                prev = curr;
                curr = 0;
            }
        }

        return max;
    }
}
