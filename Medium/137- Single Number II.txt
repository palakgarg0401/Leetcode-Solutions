// Single Number II

class Solution {
    public int singleNumber(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();

        for(int n : nums){
            map.put(n, map.getOrDefault(n, 0) + 1);
        }

        for(int val : map.keySet()) if(map.get(val) == 1) return val;
        return -1;
    }
}
