// Maximum Product Subarray

// Approach 1
class Solution {
    public int maxProduct(int[] nums) {
        int n = nums.length;
        long max = Integer.MIN_VALUE;
        for(int i = 0; i < n; i++){
            long product = 1;
            for(int j = i; j < n; j++){
                product *= nums[j];
                max = Math.max(max, product);
            }
        }

        return (int)max;
    }
}
