// Number of 1 Bits

// Approach 1
class Solution {
    public int hammingWeight(int n) {
        if(n == 0) return 0;

        int count = 0;
        while(n > 0) {
            if(n % 2 == 1) count++;
            n /= 2;
        }

        return count;
    }
}


// Approach 2
class Solution {
    public int hammingWeight(int n) {
        int count = 0;
        while(n > 0){
            count += n & 1;
            n = n >> 1;
        }

        return count;
    }
}
