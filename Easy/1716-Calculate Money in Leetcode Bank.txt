// Calculate Money in Leetcode Bank

class Solution {
    public int totalMoney(int n) {
        if(n <= 7) return (n * (n+1)) / 2;

        int sum = 0, a = 28;

        for(int i = 0; i < n / 7; i++){
            sum += a;
            a += 7;
        }

        int k = (n / 7) + 1;
        for(int i = 1; i <= n % 7; i++){
            sum += k;
            k++;
        }

        return sum;
    }
}
