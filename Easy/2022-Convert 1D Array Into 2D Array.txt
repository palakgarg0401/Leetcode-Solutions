//Convert 1D Array Into 2D Array

class Solution {
    public int[][] construct2DArray(int[] original, int m, int n) {
        if(m * n != original.length) return new int[][]{};

        int k = 0;
        int[][] res = new int[m][n];

        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                res[i][j] = original[k++];
            }
        }

        return res;
    }
}
