// Delete Columns to Make Sorted

class Solution {
    public int minDeletionSize(String[] strs) {
        if(strs.length < 1) return 0;
        int n = strs[0].length();
        int count = 0;
        for(int i = 0; i < n; i++){
            for(int j = 1; j < strs.length; j++){
                if(strs[j].charAt(i) < strs[j - 1].charAt(i)){
                    count++;
                    break;
                }
            }
        }
        return count;
    }
}
