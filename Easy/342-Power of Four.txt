//Power of Four

// Approach 1
class Solution {
    public boolean isPowerOfFour(int n) {
        if(n <= 0) return false;
        if(n == 1) return true;
        else if(n % 4 != 0) return false;
        return isPowerOfFour(n / 4);
    }
}

// Approach 2
class Solution {
    public boolean isPowerOfFour(int n) {
        if(n <= 0) return false;
        if((n & (n - 1)) != 0) return false;

        return (n & 0x55555555) == n;
    }
}
