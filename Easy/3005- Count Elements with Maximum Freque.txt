// Count Elements with Maximum Frequency

// Approach 1
class Solution {
    public int maxFrequencyElements(int[] nums) {
        int max = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for(int n : nums){
            map.put(n, map.getOrDefault(n, 0) + 1);
            max = Math.max(max, map.get(n));
        }

        int count = 0;
        for(int val : map.values()){
            if(val == max) count += max;
        }

        return count;
    }
}

// Approach 2
class Solution {
    public int maxFrequencyElements(int[] nums) {
        int[] freq = new int[101];
        int max = 0;

        for(int n : nums){
            freq[n]++;
            max = Math.max(max, freq[n]);
        }

        int count = 0;
        for(int f : freq){
            if(f == max) count += f;
        }

        return count;
    }
}
