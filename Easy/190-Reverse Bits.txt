// Reverse Bits

// Approach 1
class Solution {
    public int reverseBits(int n) {
        StringBuilder binary = new StringBuilder();
        for(int i = 0; i < 32; i++){
            binary.append((n & 1));
            n = n >> 1;
        }

        return (int)Long.parseLong(binary.toString(), 2);
    }
}
